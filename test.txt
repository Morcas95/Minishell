#!/usr/bin/env bash

set -euo pipefail

ROOT_DIR="$(cd "$(dirname "$0")" && pwd)"
cd "$ROOT_DIR"

if [[ ! -x ./minishell ]]; then
	echo "[INFO] Compilando minishell..."
	make >/dev/null
fi

run_case() {
	local name="$1"
	local input="$2"
	local expect="$3"
	local output

	output="$(printf "%b" "$input" | ./minishell 2>&1 || true)"
	if echo "$output" | grep -Eq "$expect"; then
		echo "[OK] $name"
	else
		echo "[FAIL] $name"
		echo "--- output ---"
		echo "$output"
		echo "--------------"
		return 1
	fi
}

echo "[INFO] Ejecutando smoke tests..."

run_case "pwd builtin" \
	"pwd\nexit\n" \
	"/home/diego/minishell"

run_case "echo builtin" \
	"echo hola mundo\nexit\n" \
	"hola mundo"

run_case "export + env" \
	"export TEST_MINI=42\nenv | grep TEST_MINI\nexit\n" \
	"TEST_MINI=42"

run_case "unset" \
	"export TEST_MINI=42\nunset TEST_MINI\nenv | grep TEST_MINI\nexit\n" \
	"Exit status: 1"

run_case "cd persists in parent" \
	"cd /\npwd\nexit\n" \
	"^/$"

run_case "external via PATH" \
	"ls / | head -n 1\nexit\n" \
	"Exit status: 0"

run_case "pipeline with builtin + external" \
	"echo abc | wc -c\nexit\n" \
	"4"

echo "[INFO] Todo OK"
